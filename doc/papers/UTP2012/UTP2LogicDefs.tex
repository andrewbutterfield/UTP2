\def\INFERENCES{

     \INFER{P}{P[Q:=R]}

  \\
  \\ \INFER{P\equiv Q}{R[S:=P] \equiv R[S:=Q]}

  \\ \INFER{P, P \equiv Q}{Q}

  \end{array}

\def\DERIVED{
\begin{array}{ll}
   \INFER{P\equiv Q , Q \equiv R}{P \equiv R}
 & \mbox{(Transitivity)}
%
\\ \Gamma \vdash p
   \quad\mbox{iff}\quad
   \Gamma \vdash p \equiv True
 & \mbox{(Truth-is-True)}
%
\\\hline\vspace{-9pt}
\\ \vdash (\forall \qsep\vec x\qsep\vec q @ True) \equiv True
& \mbox{(Always-True)}
\\ \vdash (\forall \qsep Es @ True) \equiv True
\\ \vdash (\forall \qsep Ps @ True) \equiv True
%
\\\hline\vspace{-9pt}
\\ \Gamma \vdash p
   \quad\mbox{iff}\quad
    \Gamma \vdash \forall \qsep\vec x\qsep\vec q @ p
 & \mbox{($\forall$ Intro/Elim)}
\\ \Gamma \vdash p
   \quad\mbox{iff}\quad
    \Gamma \vdash \forall \qsep Es @ p
\\ \Gamma \vdash p
   \quad\mbox{iff}\quad
    \Gamma \vdash \forall \qsep Ps @ p
%
\\\hline\vspace{-9pt}
\\ p \vdash \forall \qsep\vec x\qsep\vec q @ p
 & \mbox{(Generalization)}
\\ p \vdash \forall \qsep Es @ p
 & p \vdash \forall \qsep Ps @ p
%
\\\hline\vspace{-9pt}
\\ p \vdash p[\vec x:=\vec e]
 & \mbox{(Corr. Generalization)}
\\ p \vdash p[Es:=\vec e]
 & p \vdash p[ Ps:=\vec q]
%
\\\hline\vspace{-9pt}
\\ \vdash p \implies (\forall \qsep\vec x,\vec y\qsep\vec q @ r)
 & \mbox{(Universal Consequent 1)}
\\ \quad \equiv
          (\forall \qsep\vec x @ p \implies \forall \qsep\vec y\qsep\vec q @ r),
          \quad \vec x \notin p
\\ \vdash p \implies (\forall \qsep Es,Es @ r)
\\ \quad \equiv
          (\forall \qsep Es @ p \implies \forall \qsep Es @ r),
          \quad Es \notin p
\\ \vdash p \implies (\forall \qsep Ps, Qs @ r)
\\ \quad \equiv
          (\forall \qsep Ps @ p \implies \forall \qsep Qs @ r),
          \quad  Ps \notin p
%
\\\hline\vspace{-9pt}
\\ \vdash p \equiv \forall \qsep\vec x @ p,
   \quad \vec x \notin p
 & \mbox{(Corr. Univ. Conseq. 1)}
\\ \vdash p \equiv \forall \qsep Es @ p,
   \quad Es \notin p
 & \vdash p \equiv \forall \qsep Ps @ p,
   \quad  Ps \notin p
%
\\\hline\vspace{-9pt}
\\ \Gamma \vdash p \implies q
   \quad\mbox{iff}\quad
   \Gamma \vdash p \implies \forall \qsep\vec x @ q,
   \quad \vec x\notin p
 & \mbox{(Universal Consequent 2)}
\\ \Gamma \vdash p \implies q
   \quad\mbox{iff}\quad
   \Gamma \vdash p \implies \forall \qsep Es @ q,
   \quad Es\notin p
\\ \Gamma \vdash p \implies q
   \quad\mbox{iff}\quad
   \Gamma \vdash p \implies \forall \qsep Ps @ q,
   \quad  Ps\notin p
%
\\\hline\vspace{-9pt}
\\ \Gamma \vdash q \implies p
   \quad\mbox{iff}\quad
   \Gamma \vdash (\exists \vec x  @ q) \implies p,
   \quad \vec x\notin p
 & \mbox{(Corr. Univ.l Conseq. 2)}
\\ \Gamma \vdash q \implies p
   \quad\mbox{iff}\quad
   \Gamma \vdash (\exists Es  @ q) \implies p,
   \quad Es\notin p
\\ \Gamma \vdash q \implies p
   \quad\mbox{iff}\quad
   \Gamma \vdash (\exists  Ps  @ q) \implies p,
   \quad  Ps\notin p
%
\\\hline\vspace{-9pt}
\\ \vdash (\forall \qsep\vec x @ p) \equiv (\forall \qsep\vec z @ p[\vec x:=\vec z]),
   \quad \vec z \mbox{ fresh}
 & \mbox{($\alpha$-Renaming)}
\\ \vdash (\forall \qsep Es @ p) \equiv (\forall \qsep Es @ p[Es:=Es]),
   \quad Es \mbox{ fresh}
\\ \vdash (\forall \qsep Ps @ p) \equiv (\forall \qsep Qs @ p[ Ps:= Qs]),
   \quad  Qs \mbox{ fresh}
  \end{array}
}

\def\DEDUCTION{
  \begin{array}{ll}
     \Gamma, p \vdash q
     \quad\mbox{then}\quad
     \Gamma \vdash p \implies q,
     \quad p \mbox{ closed}
   & \mbox{(Deduction Theorem)}
  \\ \Gamma \vdash p \implies q
     \quad\mbox{then}\quad
     \Gamma, p \vdash q
   & \mbox{(Deduc. Converse)}
  \\ \Gamma, p[\vec x:=\vec k] \vdash q[\vec x:=\vec k]
     \quad\mbox{then}\quad
     \Gamma \vdash p \implies q,
   & \mbox{(Flexible Deduction)}
  \\\quad \vec x = \fv~p, \vec k \mbox{ constants.}
  \end{array}
}
